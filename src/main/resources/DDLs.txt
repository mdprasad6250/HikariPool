CREATE TABLE "BES"."STUDENT"
(
    student_Id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    student_Name VARCHAR2(30 BYTE) NOT NULL ENABLE,
    rank VARCHAR2(30 BYTE) NOT NULL ENABLE
);

CREATE TABLE "BES"."EXAMS"
(
    exam_Id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    exam_Name VARCHAR2(30 BYTE) NOT NULL ENABLE,
    exam_Results VARCHAR2(30 BYTE) NOT NULL ENABLE
);

CREATE TABLE "BES"."BOOK"
(
    book_Id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    book_Name VARCHAR2(30 BYTE) NOT NULL ENABLE,
    book_Author VARCHAR2(30 BYTE) NOT NULL ENABLE
);


CREATE TABLE "BES"."PARENTS"
(
    parent_Id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    parent_Name VARCHAR2(30 BYTE) NOT NULL ENABLE,
    parent_Feedback VARCHAR2(80 BYTE) NOT NULL ENABLE
);


CREATE TABLE "BES"."TEACHER"
(
    teacher_Id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    teacher_Name VARCHAR2(30 BYTE) NOT NULL ENABLE,
    subject VARCHAR2(30 BYTE) NOT NULL ENABLE
);

CREATE TABLE "BES"."EVENTS"
(
    event_Id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    event_Name VARCHAR2(30 BYTE) NOT NULL ENABLE,
    event_Organizer VARCHAR2(30 BYTE) NOT NULL ENABLE
);

CREATE TABLE "BES"."SPORTS"
(
    sport_Id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    sport_Name VARCHAR2(30 BYTE) NOT NULL ENABLE,
    sport_Organizer VARCHAR2(30 BYTE) NOT NULL ENABLE
);

CREATE TABLE "BES"."MANAGEMENT"
(
    management_Id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    management_Name VARCHAR2(30 BYTE) NOT NULL ENABLE,
    management_Head VARCHAR2(30 BYTE) NOT NULL ENABLE
);
